15.2 Creating Factor

ファクタを作る

factor

x1 <- c("Dec", "Apr", "Jan", "Mar")
y1 <- factor(x1, levels = c("Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"))
y1
## [1] Dec Apr Jan Mar
## Levels: Jan Feb Mar Apr May Jun Jul Aug Sep Oct Nov Dec

levelsに存在しない要素はNA

x2 <- c("Dec", "Apr", "Jam", "Mar")
y2 <- factor(x2, levels = c("Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"))
y2
## [1] Dec  Apr  <NA> Mar 
## Levels: Jan Feb Mar Apr May Jun Jul Aug Sep Oct Nov Dec

順序のコントロール

順序指定なしだとアルファベット順

factor(x1)
## [1] Dec Apr Jan Mar
## Levels: Apr Dec Jan Mar

最初に現れた順

factor(x1, levels = unique(x1))
## [1] Dec Apr Jan Mar
## Levels: Dec Apr Jan Mar

levels

factorのlevelsを文字列ベクトルとして取得

y1 <- factor(x1, levels = c("Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"))
levels(y1)
##  [1] "Jan" "Feb" "Mar" "Apr" "May" "Jun" "Jul" "Aug" "Sep" "Oct" "Nov"
## [12] "Dec"

fct_reorder

Usage

fct_reorder(.f, .x, .fun = median, ..., .desc = FALSE)

Arguments

.f factor変数(char変数でも可)
.x funを使って順序を決定するために使う変数
.fun 集計関数(ベクトルを受けて値を返す)。デフォルトはmedian
… 集計関数に渡すパラメータ。たとえばna.rm = TRUE
.desc 順番を逆にする

わかりやすくなった

relig_summary %>%
  mutate(relig = fct_reorder(relig, tvhours)) %>%
  ggplot(aes(tvhours, relig)) + geom_point()

fct_relevel

Usage

fct_relevel(.f, ..., after = 0L)

Auguments

.f factor変数(char変数でも可)
… 集計関数に渡すパラメータ。たとえばna.rm = TRUE
after どの位置に指定した値を置くか。デフォルトは一番うしろ

一個だけ動かした

ggplot(rincome_summary, aes(age, fct_relevel(rincome, "Not applicable"))) +
  geom_point()

色と線との対応が分かりにくい

by_age <- gss_cat %>%
  filter(!is.na(age)) %>%
  count(age, marital) %>%
  group_by(age) %>%
  mutate(prop = n / sum(n)) 
ggplot(by_age, aes(age, prop, colour = marital)) +
  geom_line(na.rm = TRUE)

fct_reorder2

ファクタ.fを.xが一番大きいときの.yの順序で並び替える

ggplot(by_age, aes(age, prop, colour = fct_reorder2(marital, age, prop))) +
  geom_line() +
  labs(colour = "marital")

15.5 Modifying factor levels

支持する政党

##  5 Not str republican  3032
##  6 Ind,near rep        1791
##  7 Independent         4119

##  9 Not str democrat    3690
## 10 Strong democrat     3490

表現が一貫していなくてクソ

修正

gss_cat %>%
  mutate(partyid = fct_recode(partyid,
    "Republican, strong"    = "Strong republican",
    "Republican, weak"      = "Not str republican",
    "Independent, near rep" = "Ind,near rep",
    "Independent, near dem" = "Ind,near dem",
    "Democrat, weak"        = "Not str democrat",
    "Democrat, strong"      = "Strong democrat",
    "Other"                 = "No answer",
    "Other"                 = "Don't know",
    "Other"                 = "Other party"
  )) %>%
  count(partyid)

まとめて修正

gss_cat %>%
  mutate(partyid = fct_collapse(partyid,
    other = c("No answer", "Don't know", "Other party"),
    rep = c("Strong republican", "Not str republican"),
    ind = c("Ind,near rep", "Independent", "Ind,near dem"),
    dem = c("Not str democrat", "Strong democrat")
  )) %>%
  count(partyid)
