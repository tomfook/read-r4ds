## RMSE
measure_distance <- function(mod, data) {
  diff <- data$y - model1(mod, data)
  sqrt(mean(diff ^2))
}
measure_distance(c(7, 1.5), sim1) 
## [1] 2.665212

\[ \mathrm{RMSE} = \sqrt{\frac{1}{N}\sum_{i=1}^N \left(y_i - (a_0 + a_1 x)\right)^2} \]
